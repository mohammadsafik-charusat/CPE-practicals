// step 1: find n!  step 2: count zeroes from the end
#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter n: ";
    cin >> n;

    // step 1: calculate factorial
    long long fact = 1;
    for (int i = 1; i <= n; i++) {
        fact = fact * i;
    }
    cout << "n! = " << fact << endl;

    // step 2: keep dividing by 10 while last digit is 0
    int count = 0;
    while (fact % 10 == 0) {
        count = count + 1;
        fact = fact / 10;
    }

    cout << "Trailing zeroes = " << count << endl;
    return 0;
}
