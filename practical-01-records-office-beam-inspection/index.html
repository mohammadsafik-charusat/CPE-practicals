Two departments each keep a sorted linked list of employee IDs. For an audit, both lists must become one sorted list — by re-linking the existing nodes, not by copying data into new ones.

Idea: look at the front of both lists. Whichever front value is smaller, pull that node into the result and move that list forward. Repeat until one list runs out, then attach whatever remains of the other list — it is already sorted, so nothing more to check.

Dry run — list1 = [1,2,4], list2 = [1,3,4]

Why a dummy node?

At the very start, the result list is empty — there is no node yet to attach anything to. A dummy node is a throwaway placeholder that gives the code somewhere to attach the first real node, so there is no special-case handling for "is this the first node?" The final answer starts right after the dummy.

Show C++ code 
// merge two already-sorted linked lists by re-linking nodes
#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
};

Node* newNode(int val) {
    Node* n = new Node();
    n->data = val;
    n->next = NULL;
    return n;
}

Node* mergeTwoLists(Node* list1, Node* list2) {
    Node dummy;
    dummy.next = NULL;
    Node* tail = &dummy;

    while (list1 != NULL && list2 != NULL) {
        if (list1->data <= list2->data) {
            tail->next = list1;
            list1 = list1->next;
        } else {
            tail->next = list2;
            list2 = list2->next;
        }
        tail = tail->next;
    }

    if (list1 != NULL) tail->next = list1;
    else tail->next = list2;

    return dummy.next;
}

void printList(Node* head) {
    while (head != NULL) {
        cout << head->data << " ";
        head = head->next;
    }
    cout << endl;
}

int main() {
    Node* list1 = newNode(1);
    list1->next = newNode(2);
    list1->next->next = newNode(4);

    Node* list2 = newNode(1);
    list2->next = newNode(3);
    list2->next->next = newNode(4);

    Node* merged = mergeTwoLists(list1, list2);
    printList(merged);
    return 0;
}
Q: why re-link instead of making new nodes? the problem says "splicing existing nodes" — reusing nodes avoids copying data, saving both memory and time.
Q: what if both lists are empty? the while loop never runs, both ifs see NULL, dummy.next stays NULL — correctly returns an empty list.
Q: time complexity? O(n + m) — every node from both lists is visited exactly once.

A circular sequence means the array wraps around — after the last index comes the first index again, like numbers on a clock. Every value must be replaced by the sum of the next k values (or previous |k| if k is negative, or 0 if k is 0).

Idea: wrapping around an array is just the remainder after dividing by the array length — the % (modulo) operator does this in one step. Going forward: (i + step) % n. Going backward: (i - step + n) % n — the extra + n keeps the number positive before taking the remainder.

Dry run — code = [5,7,1,4], k = 3

Show C++ code 
// circular array decryption using modulo wraparound
#include <iostream>
using namespace std;

int main() {
    int n, k;
    cout << "Enter n: ";
    cin >> n;

    int code[100];
    cout << "Enter " << n << " numbers: ";
    for (int i = 0; i < n; i++) cin >> code[i];

    cout << "Enter k: ";
    cin >> k;

    int result[100];

    for (int i = 0; i < n; i++) {
        int sum = 0;

        if (k > 0) {
            for (int j = 1; j <= k; j++) {
                int idx = (i + j) % n;
                sum = sum + code[idx];
            }
        }
        else if (k < 0) {
            int steps = -k;
            for (int j = 1; j <= steps; j++) {
                int idx = (i - j + n) % n;
                sum = sum + code[idx];
            }
        }
        else {
            sum = 0;
        }

        result[i] = sum;
    }

    cout << "Decrypted array: ";
    for (int i = 0; i < n; i++) cout << result[i] << " ";
    cout << endl;
    return 0;
}
Q: why % n instead of manual checking? without it you would need an extra if to bring an out-of-range index back — modulo does that in one step.
Q: why add + n before % for negative steps? C++ can return a negative remainder for negative numbers; adding n first keeps the value positive so the wraparound is correct.
Q: why a separate result array? updating code[] in place would corrupt values that later indices still need to read.
Q: time complexity? O(n × |k|).

Two numbers are given as strings because they are too big for any built-in integer type. Converting to int/long long is not allowed — multiplication must happen digit by digit, exactly like the pen-and-paper method from school.

Idea: multiplying two single digits never gives more than 2 digits (max 9×9=81). So each digit-pair's product is split across exactly two positions in the result array — the position depends only on how far each digit is from the right end of its own number.

Dry run — "12" × "34"

Show C++ code 
// digit-by-digit string multiplication, no built-in big integers
#include <iostream>
#include <string>
using namespace std;

int main() {
    string num1, num2;
    cout << "Enter num1: ";
    cin >> num1;
    cout << "Enter num2: ";
    cin >> num2;

    int n1 = num1.length();
    int n2 = num2.length();
    int result[200] = {0};

    for (int i = n1 - 1; i >= 0; i--) {
        for (int j = n2 - 1; j >= 0; j--) {

            int digit1 = num1[i] - '0';
            int digit2 = num2[j] - '0';
            int mul = digit1 * digit2;

            int pos_low  = (n1 - 1 - i) + (n2 - 1 - j);
            int pos_high = pos_low + 1;

            result[pos_low] = result[pos_low] + mul;

            if (result[pos_low] >= 10) {
                result[pos_high] = result[pos_high] + result[pos_low] / 10;
                result[pos_low] = result[pos_low] % 10;
            }
        }
    }

    string answer = "";
    int size = n1 + n2;
    int i = size - 1;

    while (i > 0 && result[i] == 0) i--;
    while (i >= 0) { answer = answer + to_string(result[i]); i--; }

    cout << "Product: " << answer << endl;
    return 0;
}
Q: why size n1 + n2 for the result array? the product of an n1-digit and n2-digit number never has more than n1+n2 digits.
Q: what does num1[i] - '0' do? converts a character digit to its numeric value, using the fact that digit characters sit consecutively in ASCII.
Q: why handle carry right after each multiplication instead of at the end? both approaches work — doing it immediately keeps all the logic for one digit-pair in one place.
Q: time complexity? O(n1 × n2).
